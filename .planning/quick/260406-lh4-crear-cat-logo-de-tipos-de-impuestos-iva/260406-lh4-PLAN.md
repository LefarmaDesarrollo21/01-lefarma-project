#### Backend - Entity (TipoImpuesto.cs):
```csharp
// Based on Banco.cs pattern
public class TipoImpuesto
{
    public int IdTipoImpuesto { get; set; }
    public string Nombre { get; set; } = string.Empty;
    public string? NombreNormalizado { get; set; }
    public string Clave { get; set; } = string.Empty;  // e.g., "T16", "ISR"
    public decimal Tasa { get; set; }  // e.g., 0.16, 0.08, 0.00
    public string? Descripcion { get; set; }
    public string? DescripcionNormalizada { get; set; }
    public bool Activo { get; set; } = true;
    public DateTime FechaCreacion { get; set; }
    public DateTime? FechaModificacion { get; set; }
    
    public void Activar() => Activo = true;
    public void Desactivar() => Activo = false;
    public void ActualizarNombreNormalizado() => NombreNormalizado = Nombre?.ToLowerInvariant().RemoveDiacritics();
}
```

#### Backend - Configuration (TipoImpuestoConfiguration.cs):
- Table: "tipos_impuesto" in "catalogos" schema
- Primary key: IdTipoImpuesto (id_tipo_impuesto, value generated on add)
- Properties:
  - Nombre: nvarchar(255), required
  - NombreNormalizado: nvarchar(300)
  - Clave: nvarchar(100), required (index)
  - Tasa: decimal(18,6), required
  - Descripcion: nvarchar(1000)
  - DescripcionNormalizada: nvarchar(1000)
  - Activo: bit, default true
  - FechaCreacion: datetime, default GETUTCDATE()
  - FechaModificacion: datetime, nullable
- Indexes: ix_tipos_impuesto_nombre_normalizado, ix_tipos_impuesto_activo, ix_tipos_impuesto_clave

#### Backend - Seeding (DatabaseSeeder.cs):
Add `await SeedTiposImpuestoAsync()` after existing seed methods.
```csharp
private async Task SeedTiposImpuestoAsync()
{
    if (await _context.Set<TipoImpuesto>().AnyAsync())
        return;
    
    var tipos = new List<TipoImpuesto>
    {
        new() { IdTipoImpuesto = 1, Nombre = "IVA 16%", Clave = "T16", Tasa = 0.16m, Activo = true, FechaCreacion = DateTime.UtcNow },
        new() { IdTipoImpuesto = 2, Nombre = "IVA 8%", Clave = "T08", Tasa = 0.08m, Activo = true, FechaCreacion = DateTime.UtcNow },
        new() { IdTipoImpuesto = 3, Nombre = "IVA 0%", Clave = "T00", Tasa = 0.00m, Activo = true, FechaCreacion = DateTime.UtcNow },
        new() { IdTipoImpuesto = 4, Nombre = "Exento", Clave = "EXE", Tasa = 0.00m, Activo = true, FechaCreacion = DateTime.UtcNow },
        new() { IdTipoImpuesto = 5, Nombre = "ISR", Clave = "ISR", Tasa = 0.00m, Activo = true, FechaCreacion = DateTime.UtcNow },
        new() { IdTipoImpuesto = 6, Nombre = "Sin Impuesto", Clave = "SIN", Tasa = 0.00m, Activo = true, FechaCreacion = DateTime.UtcNow },
    };
    
    await _context.Set<TipoImpuesto>().AddRangeAsync(tipos);
    await _context.SaveChangesAsync();
}
```
